 46:     /**
 47:      * Register a binding with the container.
 48:      *
 49:      * @param  string               $abstract
 50:      * @param  Closure|string|null  $concrete
 51:      * @param  bool                 $shared
 52:      * @return void
 53:      */
 54:     public function bind($abstract, $concrete = null, $shared = false)
 55:     {
 56:         // If the given type is actually an array, we'll assume an alias is being
 57:         // defined and will grab the real abstract class name and register the
 58:         // alias with the container so it can be used as a short-cut for it.
 59:         if (is_array($abstract))
 60:         {
 61:             list($abstract, $alias) = $this->extractAlias($abstract);
 62: 
 63:             $this->alias($abstract, $alias);
 64:         }
 65: 
 66:         // If no concrete type was given, we will simply set the concrete type to
 67:         // the abstract. This allows concrete types to be registered as shared
 68:         // without being made state their classes in both of the parameters.
 69:         unset($this->instances[$abstract]);
 70: 
 71:         if (is_null($concrete))
 72:         {
 73:             $concrete = $abstract;
 74:         }
 75: 
 76:         // If the factory is not a Closure, it means it is just a class name that
 77:         // is bound into the container to an abstract type and we'll just wrap
 78:         // it up in a Closure to make things more convenient when extending.
 79:         if ( ! $concrete instanceof Closure)
 80:         {
 81:             $concrete = function($c) use ($abstract, $concrete)
 82:             {
 83:                 $method = ($abstract == $concrete) ? 'build' : 'make';
 84: 
 85:                 return $c->$method($concrete);
 86:             };
 87:         }
 88: 
 89:         $this->bindings[$abstract] = compact('concrete', 'shared');
 90:     }

212:     /**
213:      * Resolve the given type from the container.
214:      *
215:      * @param  string  $abstract
216:      * @param  array   $parameters
217:      * @return mixed
218:      */
219:     public function make($abstract, $parameters = array())
220:     {
221:         $abstract = $this->getAlias($abstract);
222: 
223:         // If an instance of the type is currently being managed as a singleton we'll
224:         // just return an existing instance instead of instantiating new instances
225:         // so the developer can keep using the same objects instance every time.
226:         if (isset($this->instances[$abstract]))
227:         {
228:             return $this->instances[$abstract]; 
229:         }
230: 
231:         $concrete = $this->getConcrete($abstract);
232: 
233:         // We're ready to instantiate an instance of the concrete type registered for
234:         // the binding. This will instantiate the types, as well as resolve any of
235:         // its "nested" dependencies recursively until all have gotten resolved.
236:         if ($this->isBuildable($concrete, $abstract))
237:         {
238:             $object = $this->build($concrete, $parameters);
239:         }
240:         else
241:         {
242:             $object = $this->make($concrete, $parameters);
243:         }
244: 
245:         // If the requested type is registered as a singleton we'll want to cache off
246:         // the instances in "memory" so we can return it later without creating an
247:         // entirely new instance of an object on each subsequent request for it.
248:         if ($this->isShared($abstract))
249:         {
250:             $this->instances[$abstract] = $object;
251:         }
252: 
253:         $this->fireResolvingCallbacks($object);
254: 
255:         return $object;

279:     /**
280:      * Instantiate a concrete instance of the given type.
281:      *
282:      * @param  string  $concrete
283:      * @param  array   $parameters
284:      * @return mixed
285:      */
286:     public function build($concrete, $parameters = array())
287:     {
288:         // If the concrete type is actually a Closure, we will just execute it and
289:         // hand back the results of the functions, which allows functions to be
290:         // used as resolvers for more fine-tuned resolution of these objects.
291:         if ($concrete instanceof Closure)
292:         {
293:             return $concrete($this, $parameters);
294:         }
295: 
296:         $reflector = new \ReflectionClass($concrete);
297: 
298:         // If the type is not instantiable, the developer is attempting to resolve
299:         // an abstract type such as an Interface of Abstract Class and there is
300:         // no binding registered for the abstractions so we need to bail out.
301:         if ( ! $reflector->isInstantiable())
302:         {
303:             $message = "Target [$concrete] is not instantiable.";
304: 
305:             throw new BindingResolutionException($message);
306:         }
307: 
308:         $constructor = $reflector->getConstructor();
309: 
310:         // If there are no constructors, that means there are no dependencies then
311:         // we can just resolve the instances of the objects right away, without
312:         // resolving any other types or dependencies out of these containers.
313:         if (is_null($constructor))
314:         {
315:             return new $concrete;
316:         }
317: 
318:         $parameters = $constructor->getParameters();
319: 
320:         // Once we have all the constructor's parameters we can create each of the
321:         // dependency instances and then use the reflection instances to make a
322:         // new instance of this class, injecting the created dependencies in.
323:         $dependencies = $this->getDependencies($parameters);
324: 
325:         return $reflector->newInstanceArgs($dependencies);
326:     }

328:     /**
329:      * Resolve all of the dependencies from the ReflectionParameters.
330:      *
331:      * @param  array  $parameters
332:      * @return array
333:      */
334:     protected function getDependencies($parameters)
335:     {
336:         $dependencies = array();
337: 
338:         foreach ($parameters as $parameter)
339:         {
340:             $dependency = $parameter->getClass();
341: 
342:             // If the class is null, it means the dependency is a string or some other
343:             // primitive type which we can not resolve since it is not a class and
344:             // we'll just bomb out with an error since we have no-where to go.
345:             if (is_null($dependency))
346:             {
347:                 $dependencies[] = $this->resolveNonClass($parameter);
348:             }
349:             else
350:             {
351:                 $dependencies[] = $this->make($dependency->name);
352:             }
353:         }
354: 
355:         return (array) $dependencies;
356:     }
