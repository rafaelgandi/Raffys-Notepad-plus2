256:     protected function parseView($view)
257:     {
258:         if (is_string($view)) return array($view, null);
259: 
260:         // If the given view is an array with numeric keys, we will just assume that
261:         // both a "pretty" and "plain" view were provided, so we will return this

264:         {
265:             return $view;
266:         }
267: 
268:         // If the view is an array, but doesn't contain numeric keys, we will assume
269:         // the the views are being explicitly specified and will extract them via
270:         // named keys instead, allowing the developers to use one or the other.
271:         elseif (is_array($view))
272:         {
273:             return array(
274:                 array_get($view, 'html'), array_get($view, 'text')
275:             );
276:         }
277:         
278:         throw new \InvalidArgumentException("Invalid view.");
